<?php
    class User {
        private $id;
        private $name;
        private $email;

        // Construtor
        public function __construct($id, $name, $email) {
            $this->id = $id;
            $this->name = $name;
            $this->email = $email;
        }

        // Métodos getters e setters
        public function getId() {
            return $this->id;
        }

        public function getNOme() {
            return $this->name;
        }

        public function setName($name) {
            $this->name = $name;
        }

        public function getEmail() {
            return $this->email;
        }

        public function setEmail($email) {
            $this->email = $email;
        }
    }
?>
